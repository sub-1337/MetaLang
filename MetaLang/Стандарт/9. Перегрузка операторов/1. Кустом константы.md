IPAddr literal<#ip>(var<string> input)
{
    var<IPAddr> result = @default; // Не присваивать значение
    var<arr> a = input.split(".");
    if a.count() != 4
        error("Wrong litearal");
    var<int> i = 0;
    foreach octet in a
    {
        result.setOctet(int(octet), i);
        i++;
    }
    return result;

}
